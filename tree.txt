JPanel panel = new JPanel();

		panel.setLayout(new java.awt.GridLayout());
		
		JScrollPane jScrollPane1 = new JScrollPane();
		
		DefaultTreeCellRenderer renderer = new DefaultTreeCellRenderer();
      renderer.setLeafIcon(IconFactory.getIconByName("folder.png"));
      renderer.setOpenIcon(IconFactory.getIconByName("folder.png"));
      mTree.setCellRenderer(renderer);
		
      mTree.setShowsRootHandles(true);
		mTree.getSelectionModel().setSelectionMode(TreeSelectionModel.SINGLE_TREE_SELECTION);
		mTree.addKeyListener(new KeyAdapter() {

			@Override
			public void keyPressed(KeyEvent pE) {

				if(pE.getKeyCode() != KeyEvent.VK_ENTER) return;
				
				DefaultMutableTreeNode node = (DefaultMutableTreeNode) mTree.getLastSelectedPathComponent();
				if(node == null) return;
				
				Object nodeInfo = node.getUserObject();
				mLog.info(nodeInfo);

			}
			
		});
		
		DefaultMutableTreeNode node = new DefaultMutableTreeNode("/");
		DefaultTreeModel model = new DefaultTreeModel(node);
		mTree.setModel(model);

      jScrollPane1.setViewportView(mTree);
      panel.add(jScrollPane1);
      
		DefaultMutableTreeNode node2  = new DefaultMutableTreeNode("/home");
		DefaultMutableTreeNode node3  = new DefaultMutableTreeNode("/home/petros");
		node.add(node2);
		node2.add(node3);
		
		model.reload();
		mTree.expandRow(1);
		mTree.expandRow(2);
		
		
		JPopupMenu popup = new JPopupMenu();
		
		popup.setPreferredSize(new Dimension(300, 200));
		popup.add(panel);
		
		popup.show(mPanelController.getCurrentPanel(), 0, 0);
		mTree.requestFocusInWindow();
